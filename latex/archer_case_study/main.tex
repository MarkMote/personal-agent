% ── Source footnotes ────────────────────────────────────────
% Horizontal rule + left-aligned citation below, like thesis style.
% Usage: \source{text} or \sourceurl{display text}{url}
\newcommand{\source}[1]{%
  \begin{tikzpicture}[remember picture, overlay]%
    \draw[slate400, line width=0.4pt] ([xshift=12mm, yshift=9mm]current page.south west) -- ([xshift=-12mm, yshift=9mm]current page.south east);%
    \node[anchor=north west, font=\tiny\color{slate400}, inner sep=0pt, text width=\paperwidth-36mm] at ([xshift=12mm, yshift=8mm]current page.south west) {#1};%
  \end{tikzpicture}%
}
\newcommand{\sourceurl}[2]{%
  \begin{tikzpicture}[remember picture, overlay]%
    \draw[slate400, line width=0.4pt] ([xshift=12mm, yshift=9mm]current page.south west) -- ([xshift=-12mm, yshift=9mm]current page.south east);%
    \node[anchor=north west, font=\tiny\color{slate400}, inner sep=0pt, text width=\paperwidth-36mm] at ([xshift=12mm, yshift=8mm]current page.south west) {\href{#2}{#1}};%
  \end{tikzpicture}%
}
