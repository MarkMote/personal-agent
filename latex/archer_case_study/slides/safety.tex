% ── Safety-Critical Control ──────────────────────────────
\begin{frame}{Safety}
  \small
  \textbf{Given:} Input-constrained dynamical system, with \textbf{state} $x \in X \subseteq \mathbb{R}^n$
  and \textbf{input} $u \in U \subset \mathbb{R}^m$

  \vskip-8pt
  \[ \dot{x} = F(x,\, u) \]
  \vskip2pt
  \textbf{Safety} is defined  \textit{from}  constraints on the state:
  \vskip-4pt
  \[ \varphi_i(x) \geq 0, \quad i = 1, \ldots, M \]
  \vskip2pt
  States satisfying all constraints form the \textbf{constraint set} (\textit{allowable set}):
  \vskip-4pt
  \[ \mathcal{C}_A := \{x \in X \mid \varphi_i(x) \geq 0, \;\; i = 1, \ldots, M\} \]
  \vskip4pt
  Example: $\mathcal{C}_A$ specifies "no collisions"
  \vskip4pt
  % \textbf{Not all states in the constraint set are safe!}
\end{frame}

% ── Invariance and Safety ────────────────────────────────
\begin{frame}{Safety of a Closed Loop System}
  \small
  Safety means satisfying constraints \textbf{for all time}.

  \vskip6pt
  \begin{block}{Forward Invariance}
    \vskip2pt
    A closed set $S \subset X$ is \textbf{forward invariant} w.r.t.\ a closed-loop system if:
    \vskip-6pt
    \[ x(t_0) \in S \;\;\Rightarrow\;\; \forall\, t \geq t_0,\;\; x(t) \in S \]
    \vspace{-10pt}
  \end{block}

  \vskip4pt
  \textbf{Safety:}\quad If $u: X \to U$ renders the closed-loop system forward invariant in $\mathcal{C}_S \subseteq \mathcal{C}_A$:
  \vskip4pt
  \begin{itemize}
    \item $u$ is a \textbf{safe control law}
    \item $\mathcal{C}_S$ is the \textbf{safe set}
    \item $x \in \mathcal{C}_S$ is a \textbf{safe state}
  \end{itemize}
  \source{\hyperlink{backup:safety-formal}{Full formal definitions in backup slides}}
\end{frame}

% ── Control Invariance ─────────────────────────────────────
\begin{frame}{Control Invariance}
  \small
  % So far: given a control law, we can tell if its safe
  % \vskip6pt
  Control invariance tells you if/where its \textbf{possible} to find a safe control law:

  \vskip6pt
  \begin{block}{Control Invariance}
    \vspace{4pt}
    A closed set $S \subset X$ is \textbf{control invariant} with respect to a dynamical system and input set $U$ if there exists a control law $u: X \to U$ that renders $S$ forward invariant.
    \vspace{4pt}
  \end{block}

  \vskip8pt
  \textbf{Viability Kernel} is the largest control invariant subset.

  \vskip0pt
  \begin{itemize}
    \item This is the "best" safe set we can find
  \end{itemize}
\end{frame}

% ── Barrier Functions ──────────────────────────────────────
\begin{frame}{SOTA: Control Barrier Functions}
  \small
  Define the safe set via a \textbf{barrier function} $h$:
  \vskip-4pt
  \[ \mathcal{C}_S = \{x \in \mathbb{R}^n \mid h(x) \geq 0\} \subseteq \mathcal{C}_A \]
  \vskip2pt
  The set of \textbf{safe inputs} at state $x$:
  \vskip-6pt
  \[ U_s(x) = \{u \in U \mid L_f h(x) + L_g h(x)\, u + \alpha\big(h(x)\big) \geq 0\} \]
  \vskip2pt
  This ensures the system is ``pulled back'' ($\dot{h}\geq0$) near the boundary of $\mathcal{C}_S$.

  \vskip6pt
  \begin{block}{Active Safety Filter (QP)}
    \vskip-6pt
    \[ u_{\text{act}} = \arg\min_{u \in U} \| u_{\text{des}} - u \|^2 \quad \text{s.t.} \quad u \in U_s(x) \]
    \vspace{-8pt}
  \end{block}
  \vskip4pt
  \textbf{If $\mathcal{C}_S$ is control invariant, the QP is always feasible.}
  \source{\hyperlink{backup:cbf-detail}{This is a simplification. More details in backup slides}}
\end{frame}

% ── Implicit Methods ──────────────────────────────────────
\begin{frame}{Implicit Safe Sets}
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \smaller
      \textbf{Key insight:} If you can reach \textit{any} safe invariant set, you are in a control invariant set.

      \vskip6pt
      Replace ``find $\mathcal{C}_S$ offline'' with:
      \vskip2pt
      \begin{itemize}
        \item \textbf{Find} a small safe set  $\mathcal{C}_b$ and a backup controller $u_b$
        \item \textbf{Simulate} from $x_0$ to $\mathcal{C}_b$ under $u_b$
        \item \textbf{Check:} trajectory safe $\Rightarrow$ $x_0$ safe
        % \item If yes $\Rightarrow$ current state is safe
      \end{itemize}

      % \vskip6pt
      % All you need:
      % \vskip2pt
      % \begin{itemize}
      %   \item A backup set $\mathcal{C}_b$ (small, easy to find)
      %   \item A backup controller
      % \end{itemize}

      \vskip6pt
      $\mathcal{C}_b$ is a safety kernel. $\mathcal{C}_s$ is \textbf{implicitly} defined by trajectories under $u_b$
      \vskip6pt

    \end{column}

% ── Online Safety (Safe Backwards Image) ─────────────────
\begin{frame}{Online Safety --- Safe Backwards Image}
  \hypertarget{backup:online-safety}{}
  \small
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      Given domain $X \subset \mathbb{R}^n$, convex input set $U \subset \mathbb{R}^m$, define:

      \vskip4pt
      $\mathcal{C}_A = \{x \mid \varphi(x) \geq 0\}$ \hfill Constraint Set

      $\mathcal{C}_b = \{x \mid h_b(x) \geq 0\}$ \hfill Invariant Backup Set

      $u_b: X \to U$ \hfill Backup Control Law

      $\phi_t^{u_b}(t;\, x_0): X \to X$ \hfill Flow under $u_b$ after time $t$

      \vskip14pt
      \textbf{Safe Backwards Image} (SBI) of $\mathcal{C}_b$:
    \end{column}
    \begin{column}{0.42\textwidth}
      \centering
      \includegraphics[width=\textwidth]{figures/implicit-safe-sets-2.png}
    \end{column}
  \end{columns}
  \vskip4pt
  \footnotesize
  \[ \mathcal{C}_S = \big\{x \in X \;\big|\; \underbrace{h_b\big(\phi^{u_b}(T;\,x)\big) \geq 0}_{\text{Reach } \mathcal{C}_b \text{ by time } T},\;\; \underbrace{\forall t \in [0,T],\; \varphi\big(\phi^{u_b}(t;\,x)\big) \geq 0}_{\text{Stay in } \mathcal{C}_A \text{ over } [0,T]} \big\} \]
  \vskip6pt
  \small
  The SBI $\mathcal{C}_S$ is an implicitly defined \textbf{control invariant} subset of $\mathcal{C}_A$. Next: develop barrier constraints to keep the system in $\mathcal{C}_S$.
  \source{Gurriet, Mote, Ames, Feron --- ``Online approach to active set invariance,'' CDC 2018}
\end{frame}

% ── Canonical Algorithm (SBSF) ───────────────────────────
\begin{frame}{Canonical Algorithm}
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \footnotesize
      \textbf{Algorithm:} $u = \text{SBSF}(\, x,\, u_{\text{des}}\,)$
      \vskip2pt
      \scriptsize
      \begin{enumerate}
        \item Take one step under $u_{\text{des}}$ to obtain
              $x_1^{\text{des}} = \phi^{u_{\text{des}}}(t_1;\, x_0)$
        \item Integrate dynamics $N$ steps under backup control law to obtain
              $\phi^{u_b}(t_i;\, x_1^{\text{des}})$\; $i = 1, \ldots, N$
        \item \textbf{if}:
              \begin{enumerate}
                \item[i.] $\phi^{u_b}(t_i;\, x_1^{\text{des}}) \in \mathcal{C}_A$ for $i = 1, \ldots, N{-}1$
                \item[ii.] $\phi^{u_b}(t_N;\, x_1^{\text{des}}) \in \mathcal{C}_b$
              \end{enumerate}
              \quad return $u_{\text{des}}$\\
              \textbf{else}:\\
              \quad return $u_b$
      \end{enumerate}

      \vskip4pt
      \footnotesize
      \textbf{Key idea:} take ``probe step'' under $u_{\text{des}}$ and test if the resulting state is safe by simulating $u_b(x)$
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{center}
        \includegraphics[width=\textwidth]{figures/sbsf.png}
      \end{center}
      \vskip-4pt
      {\scriptsize\centering Simulation-Based Switching Filter (SBSF)\par}
    \end{column}
  \end{columns}
\end{frame}

% ── Online Set Invariance (IASIF) ────────────────────────
\begin{frame}{Online Set Invariance --- IASIF}
  \hypertarget{backup:iasif}{}
  \smaller
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \smaller
      \textbf{IASIF-QP:}
      \vskip-15pt
      \smaller
      \begin{align*}
        u_{\text{act}}(x_0) &= \arg\min_{u \in U} \| u_{\text{des}}(x) - u \|^2 \\
        \text{s.t.}\quad & BC^b(x,\,u) \geq 0, \\
        & BC_t^S(x,\,u) \geq 0, \quad t = [0, t_1, \ldots, t_{N-1}]
      \end{align*}

      \vskip6pt
      \textbf{At each evaluation:}
      \smaller

      \vskip4pt
      \begin{itemize}
        \item Integrate under $u_b(x)$ for backup trajectory 
        \item Integrate $n \times n$ Jacobian: $D\phi_t^{u_b}(x_0)$
        \item Solve QP with $N$ affine constraints
      \end{itemize}
      \vskip2pt
      \small
      {\scriptsize\textbf{Key idea:} Barrier constraints along trajectory via sensitivity.}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{center}
        \includegraphics[width=\textwidth]{figures/iasif_diagram.png}
      \end{center}
    \end{column}
  \end{columns}
  \source{Gurriet, Mote, Singletary, Nilsson, Feron, Ames --- ``Scalable Safety-Critical Control,'' IEEE Access 2020}
\end{frame}

% ── Example: Segway ──────────────────────────────────────
\begin{frame}{Application: Segway}
  \hypertarget{backup:segway}{}
  \small
  \textbf{What backup controller maximizes permissiveness?}

  \vskip4pt
  \begin{block}{Optimal Control Problem (OCP)}
    \vspace{0pt}
    \footnotesize
    \vskip-8pt
    \begin{align*}
      u^*(x) &= \arg\min_{u \in U} \;\; t_f + \int_0^{t_f} x^T Q x + R u^2 \, dt \\
      \text{s.t.}\quad & \dot{x} = f(x) + g(x)\,u \\
      & x(0) = x_0,\;\; x(t_f) = 0 \\
      & x \in S
    \end{align*}
    \vspace{-12pt}
  \end{block}

  \vskip2pt
  \footnotesize
  \textbf{Segway:} Complex dynamics; OCP not tractable online.
  \textbf{Idea:} Train a backup controller from offline OCP evaluations.
  \source{Gurriet, Mote, Singletary, Feron, Ames --- ``Scalable controlled set invariance,'' CDC 2019}
\end{frame}

% ── Summary ──────────────────────────────────────────────
\begin{frame}{Summary: Runtime Assurance}
  \small
  \begin{itemize}
    \item Safety = forward invariance in a constraint-satisfying set
    \item RTA decouples safety from the primary controller: any controller is safe if the filter can always recover
    \item Implicit methods avoid the hard problem of finding large safe sets by simulating recovery online
  \end{itemize}
\end{frame}
