% ── Safety-Critical Control ──────────────────────────────
\begin{frame}{Safety}
  \small
  \textbf{Given:} Input-constrained dynamical system, with \textbf{state} $x \in X \subseteq \mathbb{R}^n$
  and \textbf{input} $u \in U \subset \mathbb{R}^m$

  \vskip-8pt
  \[ \dot{x} = F(x,\, u) \]
  \vskip2pt
  \textbf{Safety} is defined  \textit{from}  constraints on the state:
  \vskip-4pt
  \[ \varphi_i(x) \geq 0, \quad i = 1, \ldots, M \]
  \vskip2pt
  States satisfying all constraints form the \textbf{constraint set} (\textit{allowable set}):
  \vskip-4pt
  \[ \mathcal{C}_A := \{x \in X \mid \varphi_i(x) \geq 0, \;\; i = 1, \ldots, M\} \]
  \vskip4pt
  Example: $\mathcal{C}_A$ specifies "no collisions"
  \vskip4pt
  % \textbf{Not all states in the constraint set are safe!}
\end{frame}

% ── Invariance and Safety ────────────────────────────────
\begin{frame}{Safety of a Closed Loop System}
  \small
  Safety means satisfying constraints \textbf{for all time}.

  \vskip6pt
  \begin{block}{Forward Invariance}
    \vskip2pt
    A closed set $S \subset X$ is \textbf{forward invariant} w.r.t.\ a closed-loop system if:
    \vskip-6pt
    \[ x(t_0) \in S \;\;\Rightarrow\;\; \forall\, t \geq t_0,\;\; x(t) \in S \]
    \vspace{-10pt}
  \end{block}

  \vskip4pt
  \textbf{Safety:}\quad If $u: X \to U$ renders the closed-loop system forward invariant in $\mathcal{C}_S \subseteq \mathcal{C}_A$:
  \vskip4pt
  \begin{itemize}
    \item $u$ is a \textbf{safe control law}
    \item $\mathcal{C}_S$ is the \textbf{safe set}
    \item $x \in \mathcal{C}_S$ is a \textbf{safe state}
  \end{itemize}
  \source{\hyperlink{backup:safety-formal}{Full formal definitions in backup slides}}
\end{frame}

% ── Control Invariance ─────────────────────────────────────
\begin{frame}{Control Invariance}
  \small
  % So far: given a control law, we can tell if its safe
  % \vskip6pt
  Control invariance tells you if/where its \textbf{possible} to find a safe control law:

  \vskip6pt
  \begin{block}{Control Invariance}
    \vspace{4pt}
    A closed set $S \subset X$ is \textbf{control invariant} with respect to a dynamical system and input set $U$ if there exists a control law $u: X \to U$ that renders $S$ forward invariant.
    \vspace{4pt}
  \end{block}

  \vskip8pt
  \textbf{Viability Kernel} is the largest control invariant subset.

  \vskip0pt
  \begin{itemize}
    \item This is the "best" safe set we can find
  \end{itemize}
\end{frame}

% ── Barrier Functions ──────────────────────────────────────
\begin{frame}{SOTA: Control Barrier Functions}
  \small
  Define the safe set via a \textbf{barrier function} $h$:
  \vskip-4pt
  \[ \mathcal{C}_S = \{x \in \mathbb{R}^n \mid h(x) \geq 0\} \subseteq \mathcal{C}_A \]
  \vskip2pt
  The set of \textbf{safe inputs} at state $x$:
  \vskip-6pt
  \[ U_s(x) = \{u \in U \mid L_f h(x) + L_g h(x)\, u + \alpha\big(h(x)\big) \geq 0\} \]
  \vskip2pt
  This ensures the system is ``pulled back'' ($\dot{h}\geq0$) near the boundary of $\mathcal{C}_S$.

  \vskip6pt
  \begin{block}{Active Safety Filter (QP)}
    \vskip-6pt
    \[ u_{\text{act}} = \arg\min_{u \in U} \| u_{\text{des}} - u \|^2 \quad \text{s.t.} \quad u \in U_s(x) \]
    \vspace{-8pt}
  \end{block}
  \vskip4pt
  \textbf{If $\mathcal{C}_S$ is control invariant, the QP is always feasible.}
  \source{\hyperlink{backup:cbf-detail}{This is a simplifiation. More details in backup slides}}
\end{frame}

% ── The Challenge: Finding Large Safe Sets ─────────────────
\begin{frame}{The Challenge: Finding Large Safe Sets}
  \begin{center}
    \includegraphics[width=0.80\textwidth]{figures/invariant_sets_examples.png}
  \end{center}
  \vskip-4pt
  \small
  \begin{itemize}
    \item Finding \textit{some} safe invariant set is easy --- just stop, hover, or park
    \item Useful safe sets are \textbf{large} --- they let the system actually operate
    \item Large invariant sets are \textbf{hard to find} --- must explicitly characterize all states the system can remain in
  \end{itemize}
\end{frame}

% ── Implicit Methods ──────────────────────────────────────
\begin{frame}{Implicit Safe Sets}
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \small
      \textbf{Key insight:} If you can reach \textit{any} known invariant set, you are in a control invariant set.

      \vskip6pt
      Replace ``find $\mathcal{C}_S$ offline'' with:
      \vskip2pt
      \begin{itemize}
        \item Simulate forward online
        \item Check: can a \textbf{backup controller} drive the system into a \textbf{backup set} $\mathcal{C}_b$?
        \item If yes $\Rightarrow$ current state is safe
      \end{itemize}

      \vskip6pt
      All you need:
      \vskip2pt
      \begin{itemize}
        \item A backup set $\mathcal{C}_b$ (small, easy to find)
        \item A backup controller
      \end{itemize}

      \vskip6pt
      Worst case: follow the last safe trajectory.
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{center}
        \includegraphics[width=\textwidth]{figures/implicit-safe-sets.png}
      \end{center}
    \end{column}
  \end{columns}
\end{frame}
