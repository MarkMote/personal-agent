% ── Backup: Formal Safety Definitions ─────────────────────
\begin{frame}{Formal Safety Definitions}
  \hypertarget{backup:safety-formal}{}
  \small
  \begin{block}{Definition of Safety}
    \vspace{4pt}
    A state $x \in X$ is \textbf{safe} with respect to a dynamical system, an admissible control law $u: X \to U$, and a constraint set $\mathcal{C}_A$ if $x$ is in a \textbf{forward invariant} subset of $\mathcal{C}_A$. Furthermore, $u$ is said to be a \textbf{safe control law} inside that set.
    \vspace{4pt}
  \end{block}

  \vskip4pt
  \sourceurl{Mote, 2021 --- PhD Dissertation, Georgia Tech}{/papers/MOTE-DISSERTATION-2021.pdf}
\end{frame}

% ── Backup: Control Barrier Functions ─────────────────────
\begin{frame}{Control Barrier Functions --- Detailed Formulation}
  \hypertarget{backup:cbf-detail}{}
  \small
  Assume control-affine dynamics: $\dot{x} = f(x) + g(x)\,u$

  \vskip4pt
  Define the safe set as the super-level set of $h$:\quad $\mathcal{C}_S = \{x \in \mathbb{R}^n \mid h(x) \geq 0\}$

  \vskip4pt
  \begin{block}{Nagumo's Theorem}
    \vspace{2pt}
    Forward invariance of $\mathcal{C}_S$ is guaranteed if $\dot{h}(x) \geq 0$ for all $x \in \partial\mathcal{C}_S$.
    \vspace{2pt}
  \end{block}

  \vskip4pt
  Expanding $\dot{h}$ via the chain rule (assuming $L_g h(x) \neq 0$, i.e.\ relative degree 1):
  \vskip-6pt
  \[ \dot{h}(x) = \nabla h^T \dot{x} = \nabla h^T [f(x) + g(x)\,u] = L_f h(x) + L_g h(x)\, u \]
  \vskip0pt
  The set of safe inputs is then:
  \vskip-6pt
  \[ U_s(x) = \{u \in U \mid L_f h(x) + L_g h(x)\, u + \alpha\big(h(x)\big) \geq 0\} \]
  \vskip0pt
  where $\alpha$ is an extended class $\mathcal{K}$ function. This yields the \textbf{ASIF-QP}:
  \vskip-6pt
  \[ u_{\text{act}} = \arg\min_{u \in U} \| u_{\text{des}} - u \|^2 \quad \text{s.t.} \quad u \in U_s(x) \]

  \sourceurl{Ames et al., ``Control Barrier Functions: Theory and Applications,'' ECC 2019}{https://hybrid-robotics.berkeley.edu/publications/ECC2019_Tutorial_CBFs.pdf}
\end{frame}

% ── Backup: CWH In-Plane Solution ────────────────────────
\begin{frame}{CWH In-Plane Analytical Solution}
  \hypertarget{backup:cwh-solution}{}
  \footnotesize
  The in-plane ($x$-$y$) CWH equations with $u = 0$ have closed-form solutions:
  \vskip-4pt
  \scriptsize
  \begin{align*}
    s_x(t) &= \left(4 - 3\cos(nt)\right) s_{x_0} + \frac{\sin(nt)}{n} v_{x_0} + \frac{2}{n}\left(1 - \cos(nt)\right) v_{y_0} \\
    s_y(t) &= 6\left(\sin(nt) - nt\right) s_{x_0} + s_{y_0} - \frac{2}{n}\left(1 - \cos(nt)\right) v_{x_0} + \frac{4\sin(nt) - 3nt}{n} v_{y_0}
  \end{align*}
  \vskip0pt
  \footnotesize
  \textbf{Closed (periodic) condition:} The $s_y$ secular drift term $-6nt\, s_{x_0} - 3nt\, v_{y_0}/n$ vanishes iff:
  \vskip-6pt
  \[ v_{y_0} = -2n\, s_{x_0} \]
  \vskip0pt
  \textbf{CNMT conditions:} Adding the elliptical orbit condition $v_{x_0} = 0.5n\, s_{y_0}$:
  \vskip-6pt
  \[ M = \{x \in \mathbb{R}^4 \mid v_x = 0.5n\, s_y,\;\; v_y = -2n\, s_x\} \]
  \vskip0pt
  Each trajectory in $M$ traces a 2:1 ellipse with semi-major axis along $y$ and period $\tau = 2\pi/n$.
  \source{Clohessy \& Wiltshire, 1960;\quad Mote et al., IEEE Aerospace 2021}
\end{frame}

% ── Backup: Mixed Monotonicity for RSO ───────────────────
\begin{frame}{Mixed Monotonicity --- Reachable Set Overapproximation}
  \hypertarget{backup:mixed-monotonicity}{}
  \footnotesize
  \textbf{Goal:} Bound all possible trajectories under $w \in W$ without exhaustive simulation.

  \vskip4pt
  \textbf{Decomposition function} $d(x, \hat{x})$ satisfies for all $x, \hat{x} \in X$:
  \vskip-6pt
  \scriptsize
  \[ d(x, x) = f(x) \qquad \text{and} \qquad \frac{\partial d_i}{\partial x_j} \geq 0, \quad \frac{\partial d_i}{\partial \hat{x}_j} \leq 0 \]
  \vskip0pt
  \footnotesize
  \textbf{Embedding system} ($2n$ dimensions):
  \vskip-6pt
  \scriptsize
  \[ \dot{\underline{x}} = d(\underline{x},\, \overline{x}), \qquad \dot{\overline{x}} = d(\overline{x},\, \underline{x}) \]
  \vskip0pt
  \footnotesize
  \textbf{Result:} If $\underline{x}(0) \leq x(0) \leq \overline{x}(0)$, then for all $t \geq 0$:
  \vskip-6pt
  \[ \underline{x}(t) \leq x(t) \leq \overline{x}(t) \]
  \vskip0pt
  \textbf{Tight hyperrectangular overapproximation} of the reachable set at each timestep, from a single $2n$-dim ODE.
  \source{Coogan, ``Mixed monotonicity for reachability and safety,'' CDC 2020}
\end{frame}

\begin{frame}{Collision Model --- Experimental Derivation}
  \hypertarget{backup:collision-model}{}
  \footnotesize
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \textbf{Setup:} Free-flying spacecraft on air bearing table (3-DOF: $s_x$, $s_y$, $\theta$).

      \vskip4pt
      \textbf{Procedure:}
      \vskip2pt
      \scriptsize
      \begin{enumerate}
        \item Drive spacecraft into wall at controlled angles and speeds
        \item Record pre/post-collision velocities via motion capture
        \item Fit algebraic model via least-squares regression
      \end{enumerate}

      \vskip2pt
      \footnotesize
      \textbf{Model structure:} Post-collision state is affine in pre-collision normal and tangential velocities.

      \vskip2pt
      \textbf{Key property:} Algebraic --- integrates directly into mixed-integer optimization as linear constraints.

% ── Backup: Stabilization to NMT Set ─────────────────────
\begin{frame}{Stabilization to NMT Set}
  \hypertarget{backup:nmt-stabilization}{}
  \footnotesize
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \textbf{NMT Set:}\quad $M = \{x \in \mathbb{R}^4 \mid v_x = 0.5n\, s_y,\;\; v_y = -2n\, s_x\}$

      \vskip2pt
      \textbf{Error:}\quad $e_1 = v_x - 0.5n\, s_y$, \quad $e_2 = v_y + 2n\, s_x$

      \vskip2pt
      \textbf{Error Dynamics:}
      \vskip-6pt
      \scriptsize
      \[ \begin{bmatrix} \dot{e}_1 \\ \dot{e}_2 \end{bmatrix} = \begin{bmatrix} 0 & 1.5n \\ 0 & 0 \end{bmatrix} \begin{bmatrix} e_1 \\ e_2 \end{bmatrix} + \begin{bmatrix} u_x \\ u_y \end{bmatrix} \]
      \vskip0pt
      \footnotesize
      \textbf{Choose:}\quad $u_x = -1.5n\, e_2 - K e_1$, \quad $u_y = -K e_2$

      \vskip2pt
      \textbf{Closed Loop:}
      \vskip-6pt
      \scriptsize
      \[ \begin{bmatrix} \dot{e}_1 \\ \dot{e}_2 \end{bmatrix} = \begin{bmatrix} -K & 0 \\ 0 & -K \end{bmatrix} \begin{bmatrix} e_1 \\ e_2 \end{bmatrix} \]
    \end{column}

% ── Backup: ARPOD IASIF Constraint Details ───────────────
\begin{frame}{IASIF Constraints for ARPOD}
  \hypertarget{backup:arpod-iasif}{}
  \footnotesize
  \textbf{Backup controller:} $u_b(x) = $ NMT stabilization law (drives $e \to 0$ at rate $K$)

  \vskip2pt
  \textbf{Backup set:} $\mathcal{C}_b = \{x \mid \|e\| \leq \epsilon\} \cap \mathcal{C}_A$ \quad (near-NMT states within constraints)

  \vskip4pt
  \textbf{IASIF barrier constraints} at trajectory point $t_i$:
  \vskip-6pt
  \scriptsize
  \begin{align*}
    BC^b(x, u) &: \quad \nabla h_b(\phi_N)^T D\phi_N \cdot [F(x, u_{\text{des}}) - F(x, u_b)] \geq -\alpha(h_b(\phi_N)) \\
    BC^S_{t_i}(x, u) &: \quad \nabla \varphi_j(\phi_i)^T D\phi_i \cdot [F(x, u_{\text{des}}) - F(x, u_b)] \geq -\alpha(\varphi_j(\phi_i))
  \end{align*}
  \vskip2pt
  \footnotesize
  where $\phi_i = \phi^{u_b}(t_i;\, x)$ and $D\phi_i = D_x \phi^{u_b}(t_i;\, x)$ is the state-transition sensitivity.

  \vskip2pt
  \scriptsize
  \textbf{Parameters:} $N = 75$, $\Delta t = 1$s, $K = 0.01$ 1/s, $\alpha(x) = 10x^3$, $\epsilon = 0.05$. \quad QP with $N \!\times\! M + 1$ constraints, solved at $\sim$10 Hz.
  \source{Gurriet, Mote, Singletary, Nilsson, Feron, Ames --- IEEE Access, 2020}
\end{frame}
