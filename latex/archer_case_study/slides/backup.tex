% ── Backup: Formal Safety Definitions ─────────────────────
\begin{frame}{Formal Safety Definitions}
  \hypertarget{backup:safety-formal}{}
  \small
  \begin{block}{Definition of Safety}
    \vspace{4pt}
    A state $x \in X$ is \textbf{safe} with respect to a dynamical system, an admissible control law $u: X \to U$, and a constraint set $\mathcal{C}_A$ if $x$ is in a \textbf{forward invariant} subset of $\mathcal{C}_A$. Furthermore, $u$ is said to be a \textbf{safe control law} inside that set.
    \vspace{4pt}
  \end{block}

  \vskip4pt
  \sourceurl{Mote, 2021 --- PhD Dissertation, Georgia Tech}{/papers/MOTE-DISSERTATION-2021.pdf}
\end{frame}

% ── Backup: Control Barrier Functions ─────────────────────
\begin{frame}{Control Barrier Functions --- Detailed Formulation}
  \hypertarget{backup:cbf-detail}{}
  \small
  Assume control-affine dynamics: $\dot{x} = f(x) + g(x)\,u$

  \vskip4pt
  Define the safe set as the super-level set of $h$:\quad $\mathcal{C}_S = \{x \in \mathbb{R}^n \mid h(x) \geq 0\}$

  \vskip4pt
  \begin{block}{Nagumo's Theorem}
    \vspace{2pt}
    Forward invariance of $\mathcal{C}_S$ is guaranteed if $\dot{h}(x) \geq 0$ for all $x \in \partial\mathcal{C}_S$.
    \vspace{2pt}
  \end{block}

  \vskip4pt
  Expanding $\dot{h}$ via the chain rule (assuming $L_g h(x) \neq 0$, i.e.\ relative degree 1):
  \vskip-6pt
  \[ \dot{h}(x) = \nabla h^T \dot{x} = \nabla h^T [f(x) + g(x)\,u] = L_f h(x) + L_g h(x)\, u \]
  \vskip0pt
  The set of safe inputs is then:
  \vskip-6pt
  \[ U_s(x) = \{u \in U \mid L_f h(x) + L_g h(x)\, u + \alpha\big(h(x)\big) \geq 0\} \]
  \vskip0pt
  where $\alpha$ is an extended class $\mathcal{K}$ function. This yields the \textbf{ASIF-QP}:
  \vskip-6pt
  \[ u_{\text{act}} = \arg\min_{u \in U} \| u_{\text{des}} - u \|^2 \quad \text{s.t.} \quad u \in U_s(x) \]

  \sourceurl{Ames et al., ``Control Barrier Functions: Theory and Applications,'' ECC 2019}{https://hybrid-robotics.berkeley.edu/publications/ECC2019_Tutorial_CBFs.pdf}
\end{frame}

% ── Backup: Online Safety (Safe Backwards Image) ─────────
\begin{frame}{Online Safety --- Safe Backwards Image}
  \hypertarget{backup:online-safety}{}
  \small
  \begin{columns}[c]
    \begin{column}{0.55\textwidth}
      Given domain $X \subset \mathbb{R}^n$, convex input set $U \subset \mathbb{R}^m$, define:

      \vskip4pt
      $\mathcal{C}_A = \{x \mid \varphi(x) \geq 0\}$ \hfill Constraint Set

      $\mathcal{C}_b = \{x \mid h_b(x) \geq 0\}$ \hfill Invariant Backup Set

      $u_b: X \to U$ \hfill Backup Control Law

      $\phi_t^{u_b}(t;\, x_0): X \to X$ \hfill Flow under $u_b$ after time $t$

      \vskip4pt
      \textbf{Safe Backwards Image} (SBI) of $\mathcal{C}_b$:
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{center}
        \includegraphics[width=\textwidth]{figures/implicit-safe-sets.png}
      \end{center}
    \end{column}
  \end{columns}
  \vskip-4pt
  \footnotesize
  \[ \mathcal{C}_S = \big\{x \in X \;\big|\; \underbrace{h_b\big(\phi^{u_b}(T;\,x)\big) \geq 0}_{\text{Reach } \mathcal{C}_b \text{ by time } T},\;\; \underbrace{\forall t \in [0,T],\; \varphi\big(\phi^{u_b}(t;\,x)\big) \geq 0}_{\text{Stay in } \mathcal{C}_A \text{ over } [0,T]} \big\} \]
  \vskip-6pt
  \small
  The SBI $\mathcal{C}_S$ is an implicitly defined \textbf{control invariant} subset of $\mathcal{C}_A$. Next: develop barrier constraints to keep the system in $\mathcal{C}_S$.
  \source{Gurriet, Mote, Ames, Feron --- ``Online approach to active set invariance,'' CDC 2018}
\end{frame}

% ── Backup: Online Set Invariance (IASIF) ────────────────
\begin{frame}{Online Set Invariance --- IASIF}
  \hypertarget{backup:iasif}{}
  \small
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \textbf{IASIF-QP:}
      \vskip-4pt
      \begin{align*}
        u_{\text{act}}(x_0) &= \arg\min_{u \in U} \| u_{\text{des}}(x) - u \|^2 \\
        \text{s.t.}\quad & BC^b(x,\,u) \geq 0, \\
        & BC_t^S(x,\,u) \geq 0, \quad t = [0, t_1, \ldots, t_{N-1}]
      \end{align*}

      \vskip-6pt
      \textbf{At each evaluation:}
      \vskip-2pt
      \begin{itemize}
        \item Integrate under $u_b(x)$ for backup trajectory $\phi_t(x_0)$
        \item Integrate $n \times n$ Jacobian: $D\phi_t^{u_b}(x_0)$
        \item Solve QP with $N$ affine constraints
      \end{itemize}
      \vskip2pt
      {\footnotesize\textbf{Key idea:} Barrier constraints along trajectory via sensitivity.}
    \end{column}
    \begin{column}{0.45\textwidth}

  \source{Gurriet, Mote, Singletary, Nilsson, Feron, Ames --- ``Scalable Safety-Critical Control,'' IEEE Access 2020}

% ── Backup: Example --- Segway ───────────────────────────
\begin{frame}{Example: Segway}
  \hypertarget{backup:segway}{}
  \small
  \begin{columns}[c]
    \begin{column}{0.55\textwidth}
      \textbf{What backup controller maximizes permissiveness?}

      \vskip4pt
      \begin{block}{Optimal Control Problem (OCP)}
        \vspace{0pt}
        \footnotesize
        \vskip-8pt
        \begin{align*}
          u^*(x) &= \arg\min_{u \in U} \;\; t_f + \int_0^{t_f} x^T Q x + R u^2 \, dt \\
          \text{s.t.}\quad & \dot{x} = f(x) + g(x)\,u \\
          & x(0) = x_0,\;\; x(t_f) = 0 \\
          & x \in S
        \end{align*}
        \vspace{-12pt}
      \end{block}

      \vskip4pt
      \textbf{Segway:} Complex dynamics; not tractable to solve OCP online.

      \vskip4pt
      \textbf{Idea:} Use offline evaluations of the OCP to \textbf{train a backup controller}.
    \end{column}
