% ── Motivation: Free-Flying Spacecraft ───────────────────
\begin{frame}{Motivation: When Collisions Are Unavoidable}
  \footnotesize
  \begin{columns}[c]
    \begin{column}{0.48\textwidth}
      \small
      Small robotic spacecraft perform automated {\color{blue600}monitoring and assistance} inside space stations.

      \vskip4pt
      \textbf{Problem:} Traditional planners treat all collisions as failures --- but sometimes collisions are \textbf{unavoidable}.

      \vskip4pt
      \textbf{Key question:} What is the best strategy when collision-free trajectories don't exist?

      \vskip4pt
      \footnotesize
      \textbf{Approach:} Model collisions as algebraic constraints and solve via mixed-integer optimization.

% ── Collision Model and Solution ─────────────────────────
\begin{frame}{Collision Model and Solution}
  \footnotesize
  \begin{columns}[c]
    \begin{column}{0.50\textwidth}
      \textbf{Nominal Dynamics:}\quad $\ddot{s}_x = u_x$,\quad $\ddot{s}_y = u_y$,\quad $\ddot{\theta} = u_\theta$

      \vskip4pt
      \textbf{Collision Model} (empirical, algebraic):
      \vskip-6pt
      \scriptsize
      \[
        \begin{bmatrix} v_N^{i+1} \\ v_T^{i+1} \\ \omega^{i+1} \end{bmatrix}
        = \begin{bmatrix} a_{11} & 0 \\ 0 & a_{22} \\ 0 & a_{32} \end{bmatrix}
        \begin{bmatrix} v_N^i \\ v_{\text{rel}}^i \end{bmatrix}
      \]
      \vskip0pt
      \footnotesize
      Coefficients fit from experimental data.

      \vskip2pt
      \textbf{Solution:} MIQP --- global optimality, completeness
    \end{column}

% ── Conclusion: Collisions → Feasibility → Safety ───────
\begin{frame}{Planning for Failure --- Connection to Safety}
  \small
  \begin{columns}[c]
    \begin{column}{0.55\textwidth}
      \textbf{Collision-tolerant planners can:}
      \vskip4pt
      \begin{itemize}
        \item Optimally mitigate damage via a predefined \textit{damage function}
        \item \textbf{Expand the feasible set} for trajectory planning
        \item Serve as better \textbf{backup strategies} for RTA systems
      \end{itemize}

      \vskip6pt
      \footnotesize
      \begin{center}
