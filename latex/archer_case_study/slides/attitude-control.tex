% ── Attitude Control Problem ─────────────────────────────
\begin{frame}{Attitude Control Problem}
  \small
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \textbf{Dynamics:}
      \vskip-4pt
      \footnotesize
      \[
        \dot{q} = Q(q)\,\omega \qquad
        \dot{\omega} = -J^{-1}(\omega \times J\omega) + J^{-1}(u + w)
      \]
      \vskip0pt
      \small
      where,
      \vskip2pt
      \footnotesize
      \begin{tabular}{@{}ll}
        $q \in \mathbb{H} \subset \mathbb{R}^4$ & quaternions \\
        $\omega \in \mathbb{R}^3$ & body rotation rates \\
        $u \in \mathbb{R}^3$ & body torques \\
        $w \in \mathbb{R}^3$ & \textbf{nondeterministic} input
      \end{tabular}

      \vskip4pt
      \small
      \textbf{Actuation:}\quad $u \in U = [-u_{\max},\, u_{\max}]^3$

      \vskip2pt
      \textbf{Perturbation:}\quad $w \in W = [-w_{\max},\, w_{\max}]^3$

      \vskip4pt
      \textbf{New challenge:} Unknown disturbances --- cannot simulate a single trajectory under $u_b$
    \end{column}

% ── Line-of-Sight Constraint ────────────────────────────
\begin{frame}{Line-of-Sight (LOS) Constraint}
  \small
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \textbf{Given:}
      \vskip2pt
      \footnotesize
      \begin{itemize}
        \item $\hat{b} \in \mathbb{R}^3$: {\color{blue600}boresight vector}
        \item $\hat{c} \in \mathbb{R}^3$: {\color{blue600}center axis of safe cone}
        \item $\beta_{\max} \in [0, \pi]$: {\color{blue600}maximum opening angle}
      \end{itemize}

      \vskip4pt
      \small
      \textbf{Safety Constraint:}\quad $\beta \leq \beta_{\max}$

      \vskip4pt
      \footnotesize
      Redefine frames: $\hat{b} = \hat{k}$ and $\hat{c} = \hat{K}$, then:
      \vskip-4pt
      \[ \cos(\beta) = \hat{c}^T \mathrm{R}(q)\, \hat{b} = 1 - 2q_1^2 - 2q_2^2 \]

      \vskip0pt
      \small
      \textbf{Constraint function:}
      \vskip-6pt
      \[ \varphi(x) := \cos(\beta_{\max}) - 1 + 2q_1^2 + 2q_2^2 \geq 0 \]
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{center}

        {\color{slate400}\framebox(180,160){att\_los\_cone.png}}
      \end{center}
    \end{column}
  \end{columns}
\end{frame}

% ── From Deterministic to Nondeterministic Safety ────────
\begin{frame}{Safety with Nondeterminism}
  \small
  \begin{columns}[c]
    \begin{column}{0.46\textwidth}
      \begin{center}
        \textbf{Deterministic}
        \vskip4pt
        % Save as: figures/att_deterministic.png
        {\color{slate400}\framebox(180,110){att\_deterministic.png}}
      \end{center}
      \vskip2pt
      \scriptsize
      Safe if {\color{blue600}recovery trajectory} safely reaches $\mathcal{C}_b$
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{center}
        \textbf{Nondeterministic}
        \vskip4pt
        % Save as: figures/att_nondeterministic.png
        {\color{slate400}\framebox(180,110){att\_nondeterministic.png}}
      \end{center}
      \vskip2pt
      \scriptsize
      Safe if {\color{blue600}reachable set around recovery trajectory} safely reaches $\mathcal{C}_b$
    \end{column}
  \end{columns}

  \vskip6pt
  \small
  \textbf{Key extension:} Replace trajectory simulation with \textbf{reachable set overapproximation} (RSO) to account for $w \in W$

  \vskip4pt
  \footnotesize
  RSO computed online via {\color{blue600}mixed monotonicity}: integrate a $2n$-dimensional embedding system to bound the $n$-dimensional reachable set
\end{frame}

% ── MiMoSA + Results ────────────────────────────────────
\begin{frame}{MiMoSA --- Architecture and Results}
  \footnotesize
  \begin{columns}[c]
    \begin{column}{0.48\textwidth}
      \textbf{Mixed Monotone Simplex Architecture:}
      \vskip2pt
      \scriptsize
      \begin{enumerate}
        \item $u_b$: Recovery controller (bounded, drives $\beta \to 0$)
        \item $\mathcal{X}$: RSO via mixed monotonicity
        \item $\mathcal{C}_b$: Backup set (Lyapunov + SOS)
      \end{enumerate}

      \vskip2pt
      \footnotesize
      \textbf{RTA Control Law:}
      \vskip-6pt
      \[
        u_{\text{act}} = \begin{cases} u_{\text{des}} & \text{if } \vartheta(\mathcal{X}_p) \\ u_b(x) & \text{otherwise} \end{cases}
      \]
      \vskip0pt
      \textbf{Result:} $u_{\text{act}}$ is \textbf{robustly forward invariant} in $\mathcal{C}_S \subset \mathcal{C}_A$

      \vskip4pt
      \scriptsize
      Constraint never violated despite adversarial user input and unknown disturbances.
    \end{column}
    \begin{column}{0.49\textwidth}

        \vskip2pt

    \end{column}
  \end{columns}
