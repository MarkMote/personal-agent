% ── Attitude Control Problem ─────────────────────────────
\begin{frame}{Attitude Control Problem}
  \small
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \textbf{Dynamics:}
      \vskip-4pt
      \footnotesize
      \[
        \dot{q} = Q(q)\,\omega \qquad
        \dot{\omega} = -J^{-1}(\omega \times J\omega) + J^{-1}(u + w)
      \]
      \vskip0pt
      \small
      where,
      \vskip6pt
      \footnotesize
      \begin{tabular}{@{}ll}
        $q \in \mathbb{H} \subset \mathbb{R}^4$ & quaternions \\
        $\omega \in \mathbb{R}^3$ & body rotation rates \\
        $u \in \mathbb{R}^3$ & body torques \\
        $w \in \mathbb{R}^3$ & \textbf{nondeterministic} input
      \end{tabular}

      \vskip8pt
      \small
      \textbf{Actuation:}\quad $u \in U = [-u_{\max},\, u_{\max}]^3$

      \vskip8pt
      \textbf{Perturbation:}\quad $w \in W = [-w_{\max},\, w_{\max}]^3$

      \vskip4pt
      % \textbf{New challenge:} Unknown disturbances --- cannot simulate a single trajectory under $u_b$
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{center}
        \includegraphics[width=\textwidth]{figures/att_spacecraft_frames.png}
      \end{center}
    \end{column}
  \end{columns}
\end{frame}

% ── Line-of-Sight Constraint ────────────────────────────
\begin{frame}{Line-of-Sight (LOS) Constraint}
  \small
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \textbf{Given:}
      \vskip2pt
      \smaller
      \begin{itemize}
        \item $\hat{b} \in \mathbb{R}^3$: {\color{blue600}boresight vector}
        \item $\hat{c} \in \mathbb{R}^3$: {\color{blue600}center axis of safe cone} 
        \item $\beta_{\max} \in [0, \pi]$: {\color{blue600}exclusion half-angle}
      \end{itemize}

      \vskip4pt
      \smaller
      \textbf{Safety Constraint:}\quad $\beta \leq \beta_{\max}$

      \vskip4pt
      % \footnotesize
      % Redefine frames: $\hat{b} = \hat{k}$ and $\hat{c} = \hat{K}$, then:
      % \vskip-4pt
      % \[ \cos(\beta) = \hat{c}^T \mathrm{R}(q)\, \hat{b} = 1 - 2q_1^2 - 2q_2^2 \]

      \vskip0pt
      \smaller
      % \textbf{Constraint function:}
      \vskip6pt
      By redefining frames, can be simplified to: 
      \[ \varphi(x) := \cos(\beta_{\max}) - 1 + 2q_1^2 + 2q_2^2 \geq 0 \]
      \vskip6pt
      Backup Controller: Drives system to $\beta=\dot{\beta}=0$
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{center}
        \includegraphics[width=\textwidth]{figures/att_los_cone.png}
      \end{center}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Safety with Nondeterminism}
  \vskip-14pt
  \small
  \noindent\begin{columns}[T]
    \begin{column}{0.495\textwidth}
      \centering
      \textbf{Deterministic}
      \vskip2pt
      \includegraphics[width=\textwidth, height=0.50\textheight, keepaspectratio]{figures/att_deterministic.png}
      \vskip2pt
      \scriptsize
      % Safe if {\color{blue600}recovery trajectory} safely reaches $\mathcal{C}_b$
    \end{column}
    \begin{column}{0.495\textwidth}
      \centering
      \textbf{Nondeterministic}
      \vskip2pt
      \includegraphics[width=\textwidth, height=0.50\textheight, keepaspectratio]{figures/att_nondeterministic.png}
      \vskip2pt
      \scriptsize
      % Safe if {\color{blue600}reachable set around recovery trajectory} safely reaches $\mathcal{C}_b$
    \end{column}
  \end{columns}

  \vskip18pt
  \smaller
  \textbf{Key extension:} Replace trajectory simulation with \textbf{reachable set overapproximation} (RSO)

  \vskip4pt
  \scriptsize
   RSO computed online via {\color{blue600}mixed monotonicity}: integrate corners of a bounding rectangle
  

\end{frame}

% ── MiMoSA + Results ────────────────────────────────────
\begin{frame}{Robust Online RTA}
  \footnotesize
  \begin{columns}[c]
    \begin{column}{0.48\textwidth}
      \textbf{Requirements:}
      \vskip2pt
      \scriptsize
      \begin{itemize}
        \item $u_b$: Recovery controller (bounded, drives $\beta \to 0$)
        \item $\mathcal{X}$: RSO under $u_b$ simulated online
        \item $\mathcal{C}_b$: Robustly invariant (Lyapunov + SOS)
      \end{itemize}

      \vskip12pt
      \footnotesize
      \textbf{Algorithm: Switch to $u_b$ when}
      \scriptsize
      \begin{itemize}
        \item Reach set leaves $\mathcal{C}_{A}$ or doesn't end up in $\mathcal{C}_b$
      \end{itemize}
      \vskip12pt
      \footnotesize
      \textbf{Result}: Robust forward invariance $\mathcal{C}_S$
      \scriptsize
    \end{column}
    \begin{column}{0.49\textwidth}
      \begin{center}
        \includegraphics[width=\textwidth]{figures/att_robust_safe_sets.png}
      \end{center}
      \vskip10pt
    \end{column}
  \end{columns}
  \source{Abate, Mote, Feron, Coogan --- ``Verification and RTA for dynamical systems with uncertainty,'' HSCC 2021}
\end{frame}
